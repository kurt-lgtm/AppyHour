# pipeline/pipeline_state.py

from __future__ import annotations
from dataclasses import dataclass, field
from enum import Enum

class PipelineStage(Enum):
    IDLE = 0
    INVENTORY_SYNCED = 1
    PASS1_COMPLETE = 2
    PASS2_COMPLETE = 3
    MATRIX_GENERATED = 4
    GIFTS_MERGED = 5
    FINALIZED = 6

class InvalidTransitionError(Exception):
    """Raised when attempting an illegal backward or same-stage transition."""

@dataclass
class PassProgress:
    succeeded: list[str] = field(default_factory=list)   # order IDs
    failed: list[str] = field(default_factory=list)       # order IDs
    skipped: list[str] = field(default_factory=list)      # order IDs

@dataclass
class PipelineState:
    pipeline_id: str                    # e.g. "2026-04-05-saturday"
    stage: PipelineStage = PipelineStage.IDLE
    pass1: PassProgress = field(default_factory=PassProgress)
    pass2: PassProgress = field(default_factory=PassProgress)
    matrix_path: str | None = None
    final_xlsx_path: str | None = None
    started_at: str = ""
    updated_at: str = ""
    dry_run: bool = True

    def advance(self, to: PipelineStage) -> PipelineState:
        """Return new state advanced to `to`. Raises InvalidTransitionError if backward."""
        ...

    def to_dict(self) -> dict: ...

    @classmethod
    def from_dict(cls, d: dict) -> PipelineState: ...


# pipeline/checkpoint_store.py

from pathlib import Path

class CheckpointStore:
    DEFAULT_PATH = Path(".pipeline/checkpoint.json")

    def __init__(self, path: Path = DEFAULT_PATH) -> None: ...

    def load(self) -> PipelineState | None:
        """Returns None if no checkpoint exists yet."""
        ...

    def save(self, state: PipelineState) -> None:
        """Atomic write: write to .tmp then os.replace() to avoid partial writes."""
        ...

    def backup(self, state: PipelineState) -> None:
        """Write dated backup (.pipeline/checkpoint.YYYY-MM-DD.bak) before each step."""
        ...

    def clear(self) -> None:
        """Remove checkpoint (start fresh)."""
        ...
